const video = document.getElementById("video"); const photoCanvas = document.getElementById("photoCanvas"); const photoCtx = photoCanvas.getContext("2d"); let faceImage = null; function startCamera() { navigator.mediaDevices.getUserMedia({ video: true }) .then(stream => { video.srcObject = stream; }) .catch(() => alert("カメラ使えない")); } function takePhoto() { photoCanvas.width = video.videoWidth; photoCanvas.height = video.videoHeight; photoCtx.drawImage(video, 0, 0); faceImage = new Image(); faceImage.src = photoCanvas.toDataURL("image/png"); alert("撮影OK！"); }
